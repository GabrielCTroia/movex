// movex.config.ts

import { ChatActions, ChatState, initialChatState} from './chat.types';

export default {
  resources: {
    // Register the Chat resource to Movex by assigning it a reducer
    chat: (state = initialChatState, action: ChatActions): ChatState => {
      // User Joins the Chat
      if (action.type === 'join') {
        return {
          ...state,
          userSlots: {
            ...state.userSlots,
            [action.payload.userSlot]: false,
          },
        };
      }
      // User Submits Message
      else if (action.type === 'submit') {
        const nextMsg = action.payload;

        return {
          ...state,
          messages: [...state.messages, nextMsg],
        };
      }

      return state;
    },
  },
};
